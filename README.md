# Weather Service #

Language: Java (1.8)

Framework: Spring (2.5.5)

Database: Postgres

# Properties settings #

spring.datasource.url=jdbc:postgresql://IP:PORT/your_db_name

spring.datasource.username=your_db_username

spring.datasource.password=your_db_password
